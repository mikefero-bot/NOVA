Sección 08
Código Base — Arduino IDE
Código de arranque para ESP32 con 8 botones, debounce 50ms y pantalla TFT. Copiar en Arduino IDE y grabar.
// ═══════════════════════════════════════════
// NOVA — Código base ESP32 · Arduino IDE
// Board: ESP32 Dev Module · Speed: 921600
// Lib:   TFT_eSPI (Bodmer) — Library Manager
// ═══════════════════════════════════════════
#include <TFT_eSPI.h>
#include <SPI.h>

// ── D-PAD pines ──────────────────────────
#define BTN_UP    34   // PLA azul  ▲
#define BTN_DOWN  35   // PLA azul  ▼
#define BTN_LEFT  32   // PLA azul  ◀
#define BTN_RIGHT 33   // PLA azul  ▶
// ── ABXY pines ───────────────────────────
#define BTN_A     26   // PLA coral   — Confirmar
#define BTN_B     27   // PLA ámbar   — Cancelar
#define BTN_X     14   // PLA púrpura — Especial
#define BTN_Y     13   // PLA verde   — Menú

const int PINES[] = {BTN_UP,BTN_DOWN,BTN_LEFT,BTN_RIGHT,
                      BTN_A,BTN_B,BTN_X,BTN_Y};
TFT_eSPI tft = TFT_eSPI();

// ── Debounce: true UNA vez por pulsación ─
bool presionado(int pin) {
  static unsigned long t[40]={0};
  static bool prev[40]={true};
  bool est = digitalRead(pin);
  if(est==LOW && prev[pin]==HIGH && millis()-t[pin]>50){
    prev[pin]=LOW; t[pin]=millis(); return true;
  }
  if(est==HIGH) prev[pin]=HIGH;
  return false;
}

void setup() {
  // Configurar 8 botones con pull-up interno
  for(int i=0;i<8;i++) pinMode(PINES[i], INPUT_PULLUP);
  // Inicializar pantalla TFT
  tft.init(); tft.setRotation(1);
  tft.fillScreen(TFT_BLACK);
  tft.setTextColor(TFT_GREEN, TFT_BLACK);
  tft.drawString("NOVA lista!", 10, 10, 2);
  // ⚠ Editar TFT_eSPI/User_Setup.h:
  // TFT_CS=15 · TFT_DC=2 · TFT_RST=4
  // TFT_MOSI=23 · TFT_SCLK=18
}

void loop() {
  if(presionado(BTN_UP))    { /* navegar ↑  */ }
  if(presionado(BTN_DOWN))  { /* navegar ↓  */ }
  if(presionado(BTN_LEFT))  { /* navegar ←  */ }
  if(presionado(BTN_RIGHT)) { /* navegar →  */ }
  if(presionado(BTN_A))     { /* confirmar  */ }
  if(presionado(BTN_B))     { /* cancelar   */ }
  if(presionado(BTN_X))     { /* especial   */ }
  if(presionado(BTN_Y))     { /* abrir menú */ }
}
TFT_eSPI: Editar User_Setup.h → definir pines CS=15, DC=2, RST=4, MOSI=23, SCK=18.
